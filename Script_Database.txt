CREATE TABLE Merchant(
keyId INT PRIMARY KEY IDENTITY(1,1),
merchantId INT,
merchantName nvarchar(255),
id INT)

stock_quantity NVARCHAR(255),

[price.curr] NVARCHAR(10),
[price.buynow] FLOAT,
[price.rrp] FLOAT,
[price.store] FLOAT,
[price.basePrice] FLOAT,
[text.name] NVARCHAR(255),
[uri.awTrack] NVARCHAR(255),
[uri.alternateImageTwo] NVARCHAR(255),
[uri.alternateImageThree] NVARCHAR(255),
[uri.awImage] NVARCHAR(255),
[uri.awThumb] NVARCHAR(255),
[uri.mImage] NVARCHAR(255),
[uri.mLink] NVARCHAR(255),
[vertical.id] INT,
[vertical.name] NVARCHAR(255),
[vertical.availability] INT,
[vertical.departureDate] NVARCHAR(255),
[vertical.destinationCity] NVARCHAR(255),
[vertical.destinationCountry] NVARCHAR(255),
[vertical.destinationRegion] NVARCHAR(255),
[vertical.destinationType] NVARCHAR(255),
[vertical.duration] NVARCHAR(255),
[vertical.hotelAddress] NVARCHAR(255),
[vertical.latitude] NVARCHAR(255),
[vertical.longitude] NVARCHAR(255),
[vertical.returnDate] NVARCHAR(255),
[vertical.roomType] NVARCHAR(255),
